def expects_rows(table: str, min_count: int = 1, date_col: str = "date", max_lag_days: int = 1):
    """Decorator: fail the phase if the target table has fewer than min_count rows written today."""
    import functools
    def decorator(fn):
        @functools.wraps(fn)
        def wrapper(*args, **kwargs):
            result = fn(*args, **kwargs)
            # Build count query
            if date_col:
                sql = (
                    f"SELECT COUNT(*) AS count FROM {table} "
                    f"WHERE {date_col} >= (CURRENT_DATE - INTERVAL '{max_lag_days} days')::text"
                )
            else:
                sql = f"SELECT COUNT(*) AS count FROM {table}"
            rows = query(sql)
            count = rows[0]["count"] if rows else 0
            if count < min_count:
                raise RuntimeError(
                    f"Phase wrote 0 rows to '{table}' (found {count}) — "
                    f"check API key or upstream data source"
                )
            return result
        return wrapper
    return decorator
```

"""Connection pool and psycopg2 compatibility wrappers.

IMPORTANT: Do not import this module directly.
Always use: from tools.db import get_conn, _release
This module requires DATABASE_URL to be set — guaranteed only when
imported via tools.db.__init__.py which calls load_dotenv() first.
"""
import os as _os
import psycopg2
import psycopg2.extras
import psycopg2.pool

assert _os.environ.get("DATABASE_URL"), (
    "Import via 'from tools.db import ...' — do not import tools.db.session directly. "
    "tools.db.__init__ must run load_dotenv() before this module initializes the pool."
)

_pool = None

def _get_pool():
    global _pool
    if _pool is None:
        _pool = psycopg2.pool.ThreadedConnectionPool(
            2, 30, _os.environ["DATABASE_URL"]
        )
    return _pool


class _PgCursorWrapper:
    """Cursor wrapper that auto-converts ? placeholders and SQLite date funcs."""
    def __init__(self, cursor):
        self._cur = cursor

    def execute(self, sql, params=None):
        from tools.db.query import _to_pg
        if isinstance(sql, bytes):
            self._cur.execute(sql, params or [])
        else:
            self._cur.execute(_to_pg(sql), params or [])
        return self

    def executemany(self, sql, seq):
        from tools.db.query import _to_pg
        self._cur.executemany(_to_pg(sql), seq)
        return self

    def fetchall(self): return self._cur.fetchall()
    def fetchone(self): return self._cur.fetchone()
    def __iter__(self): return iter(self._cur)
    def __enter__(self): return self
    def __exit__(self, *args): self._cur.__exit__(*args)
    def __getattr__(self, name): return getattr(self._cur, name)


class _PgConnWrapper:
    """Thin wrapper adding SQLite-compat shims to psycopg2 connection."""
    def __init__(self, conn):
        self._conn = conn

    def executescript(self, sql: str):
        from tools.db.query import _to_pg
        with self._conn.cursor() as cur:
            for stmt in sql.split(";"):
                stmt = stmt.strip()
                if stmt:
                    cur.execute(_to_pg(stmt))
        self._conn.commit()

    def execute(self, sql: str, params=None):
        from tools.db.query import _to_pg
        cur = self._conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)
        cur.execute(_to_pg(sql), params or [])
        return cur

    def cursor(self, *args, **kwargs):
        raw = self._conn.cursor(*args, **kwargs)
        return _PgCursorWrapper(raw)

    def commit(self): self._conn.commit()
    def rollback(self): self._conn.rollback()

    def close(self):
        _release(self)

    def __getattr__(self, name):
        return getattr(self._conn, name)


def get_conn() -> _PgConnWrapper:
    """Return a wrapped psycopg2 connection from the thread pool."""
    return _PgConnWrapper(_get_pool().getconn())


def _release(conn):
    """Return connection to pool."""
    try:
        raw = conn._conn if isinstance(conn, _PgConnWrapper) else conn
        _get_pool().putconn(raw)
    except Exception:
        pass
```

```python
"""Query helpers and SQLite→PostgreSQL dialect shim.

Do not import this module directly.
Always use: from tools.db import query, query_df, upsert_many, _to_pg
"""
import os as _os
import re as _re
import psycopg2.extras

assert _os.environ.get("DATABASE_URL"), (
    "Import via 'from tools.db import ...' — do not import tools.db.query directly."
)

_col_cache: dict[str, set[str]] = {}


def _invalidate_col_cache(table: str):
    _col_cache.pop(table, None)


def _pg_columns(table: str) -> set[str]:
    if table not in _col_cache:
        from tools.db.session import get_conn, _release
        conn = get_conn()
        try:
            with conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
                cur._cur.execute(
                    "SELECT column_name FROM information_schema.columns "
                    "WHERE table_schema='public' AND table_name=%s",
                    [table]
                )
                rows = cur._cur.fetchall()
                _col_cache[table] = {r["column_name"] for r in rows}
        finally:
            _release(conn)
    return _col_cache[table]


def _to_pg(sql):
    """Convert SQLite SQL to PostgreSQL-compatible SQL."""
    if isinstance(sql, bytes):
        return sql
    sql = sql.replace("?", "%s")
    sql = _re.sub(r'INTEGER\s+PRIMARY\s+KEY\s+AUTOINCREMENT', 'SERIAL PRIMARY KEY', sql, flags=_re.IGNORECASE)
    sql = sql.replace("AUTOINCREMENT", "")
    sql = _re.sub(r'INSERT\s+OR\s+REPLACE\s+INTO', 'INSERT INTO', sql, flags=_re.IGNORECASE)
    sql = _re.sub(r'INSERT\s+OR\s+IGNORE\s+INTO', 'INSERT INTO', sql, flags=_re.IGNORECASE)

    def _date_fn(m):
        arg = m.group(1).strip().strip("'\"")
        if arg.startswith('-') or arg.startswith('+'):
            try:
                n = int(arg.split()[0])
                unit = arg.split()[1].rstrip('s') if len(arg.split()) > 1 else 'day'
                if n < 0:
                    return f"(CURRENT_DATE - INTERVAL '{-n} {unit}s')::text"
                else:
                    return f"(CURRENT_DATE + INTERVAL '{n} {unit}s')::text"
            except Exception:
                pass
        return 'CURRENT_DATE::text'

    sql = _re.sub(r"date\s*\(\s*'now'\s*,\s*([^)]+)\)", _date_fn, sql, flags=_re.IGNORECASE)
    sql = _re.sub(r"date\s*\(\s*'now'\s*\)", "CURRENT_DATE::text", sql, flags=_re.IGNORECASE)
    sql = _re.sub(r"datetime\s*\(\s*'now'\s*,\s*([^)]+)\)", _date_fn, sql, flags=_re.IGNORECASE)
    sql = _re.sub(r"datetime\s*\(\s*'now'\s*\)", "NOW()::text", sql, flags=_re.IGNORECASE)
    return sql


def query(sql, params=None):
    """Execute SQL and return list of dicts."""
    from tools.db.session import get_conn, _release
    conn = get_conn()
    try:
        with conn._conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
            cur.execute(_to_pg(sql), params or [])
            return [dict(r) for r in cur.fetchall()]
    finally:
        _release(conn)


def query_df(sql, params=None):
    """Execute SQL and return a pandas DataFrame."""
    import pandas as pd
    from tools.db.session import get_conn, _release
    import sqlalchemy
    conn = get_conn()
    try:
        engine = sqlalchemy.create_engine(_os.environ["DATABASE_URL"])
        return pd.read_sql(_to_pg(sql), engine, params=params or [])
    finally:
        _release(conn)


def upsert_many(table: str, cols: list[str], rows: list, chunk_size: int = 500):
    """Batch upsert with schema-aware auto-ALTER for new columns."""
    if not rows:
        return
    from tools.db.session import get_conn, _release
    from tools.db.schema import TABLE_PKS

    conn = get_conn()
    try:
        existing = _pg_columns(table)
        new_cols = [c for c in cols if c not in existing]
        if new_cols:
            with conn._conn.cursor() as cur:
                for c in new_cols:
                    cur.execute(f"ALTER TABLE {table} ADD COLUMN IF NOT EXISTS {c} TEXT")
            conn.commit()
            _invalidate_col_cache(table)

        pk_cols = TABLE_PKS.get(table, [])
        use_cols = [c for c in cols if c in _pg_columns(table)]
        if not use_cols:
            return

        col_str = ", ".join(use_cols)
        placeholders = ", ".join(["%s"] * len(use_cols))
        if pk_cols and all(p in use_cols for p in pk_cols):
            conflict_cols = ", ".join(pk_cols)
            update_cols = [c for c in use_cols if c not in pk_cols]
            if update_cols:
                update_str = ", ".join(f"{c} = EXCLUDED.{c}" for c in update_cols)
                on_conflict = f"ON CONFLICT ({conflict_cols}) DO UPDATE SET {update_str}"
            else:
                on_conflict = f"ON CONFLICT ({conflict_cols}) DO NOTHING"
        else:
            on_conflict = "ON CONFLICT DO NOTHING"

        sql = f"INSERT INTO {table} ({col_str}) VALUES ({placeholders}) {on_conflict}"
        col_indices = [cols.index(c) for c in use_cols]
        pk_indices = [use_cols.index(p) for p in pk_cols if p in use_cols] if pk_cols else []

        for i in range(0, len(rows), chunk_size):
            batch = []
            for row in rows[i:i + chunk_size]:
                values = tuple(row[j] for j in col_indices)
                if pk_indices and any(values[j] is None for j in pk_indices):
                    continue
                batch.append(values)
            if batch:
                with conn._conn.cursor() as cur:
                    psycopg2.extras.execute_batch(cur, sql, batch)
                conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        _release(conn)
```

```python
"""BaseFetcher: base class for single-table data fetchers.

Multi-table fetchers (e.g., fetch_fmp_v2, fetch_macro) should NOT
subclass BaseFetcher — they override run() directly in their modules.

Usage:
    class MyFetcher(BaseFetcher):
        table = "my_table"

        def fetch(self) -> list[dict]:
            return api_client.get_data()

        def transform(self, raw: list[dict]) -> list[dict]:
            return [{"symbol": r["sym"], "date": r["dt"]} for r in raw]
"""
from tools.db import upsert_many


class BaseFetcher:
    table: str = ""          # must be set in subclass
    pk_cols: list = []       # auto-resolved from TABLE_PKS if empty

    def fetch(self) -> list[dict]:
        raise NotImplementedError(f"{self.__class__.__name__} must implement fetch()")

    def transform(self, raw: list[dict]) -> list[dict]:
        """Override to normalize raw API response. Default: passthrough."""
        return raw

    def run(self) -> int:
        """Fetch → transform → upsert. Returns number of rows written."""
        raw = self.fetch()
        rows = self.transform(raw)
        if not rows:
            return 0
        upsert_many(self.table, list(rows[0].keys()), rows)
        return len(rows)
```

```python
"""PipelineRunner: dependency-graph based phase orchestration.

Replaces the numbered-phase system in daily_pipeline.py.
"""
import logging
import sqlite3
import threading
import time
import traceback
from concurrent.futures import ThreadPoolExecutor, as_completed
from dataclasses import dataclass, field
from datetime import date
from typing import Callable

logger = logging.getLogger(__name__)

_checkpoint_lock = threading.Lock()  # serializes SQLite checkpoint writes


@dataclass
class Phase:
    name: str
    fn: Callable
    deps: list[str] = field(default_factory=list)
    parallel: bool = True
    min_rows: dict = field(default_factory=dict)  # table → min row count


class PipelineRunner:
    def __init__(self, phases: list[Phase], checkpoint_path: str):
        self.phases = {p.name: p for p in phases}
        self.checkpoint_path = checkpoint_path
        self._order = self._topo_sort()

    def _topo_sort(self) -> list[str]:
        """Kahn's algorithm — raises ValueError on cycle."""
        in_degree = {name: 0 for name in self.phases}
        for p in self.phases.values():
            for dep in p.deps:
                if dep not in self.phases:
                    raise ValueError(f"Phase '{p.name}' depends on unknown phase '{dep}'")
                in_degree[p.name] += 1

        # Rebuild properly
        graph = {name: [] for name in self.phases}
        in_deg = {name: 0 for name in self.phases}
        for p in self.phases.values():
            for dep in p.deps:
                graph[dep].append(p.name)
                in_deg[p.name] += 1

        queue = [n for n, d in in_deg.items() if d == 0]
        result = []
        while queue:
            node = queue.pop(0)
            result.append(node)
            for neighbor in graph[node]:
                in_deg[neighbor] -= 1
                if in_deg[neighbor] == 0:
                    queue.append(neighbor)

        if len(result) != len(self.phases):
            raise ValueError("Pipeline phase cycle detected — check deps")
        return result

    def _get_db(self):
        return sqlite3.connect(self.checkpoint_path)

    def _is_done(self, name: str) -> bool:
        if self.checkpoint_path == ":memory:":
            return False
        with _checkpoint_lock:
            conn = self._get_db()
            conn.execute("""CREATE TABLE IF NOT EXISTS pipeline_checkpoints (
                run_date TEXT, phase_name TEXT, status TEXT,
                duration_seconds REAL,
                PRIMARY KEY (run_date, phase_name))""")
            row = conn.execute(
                "SELECT 1 FROM pipeline_checkpoints WHERE run_date=? AND phase_name=? AND status='completed'",
                [date.today().isoformat(), name]
            ).fetchone()
            conn.close()
            return row is not None

    def _save(self, name: str, status: str, elapsed: float):
        if self.checkpoint_path == ":memory:":
            return
        with _checkpoint_lock:
            conn = self._get_db()
            conn.execute("""CREATE TABLE IF NOT EXISTS pipeline_checkpoints (
                run_date TEXT, phase_name TEXT, status TEXT,
                duration_seconds REAL,
                PRIMARY KEY (run_date, phase_name))""")
            conn.execute(
                "INSERT OR REPLACE INTO pipeline_checkpoints VALUES (?,?,?,?)",
                [date.today().isoformat(), name, status, elapsed]
            )
            conn.commit()
            conn.close()

    def _run_one(self, name: str) -> bool:
        phase = self.phases[name]
        if self._is_done(name):
            logger.info(f"  ⏭  {name} — already completed today, skipping")
            return True
        t0 = time.time()
        try:
            logger.info(f"  ▶ {name}")
            phase.fn()
            elapsed = time.time() - t0
            self._save(name, "completed", elapsed)
            logger.info(f"  ✓ {name} completed in {elapsed:.1f}s")
            return True
        except Exception as e:
            elapsed = time.time() - t0
            self._save(name, "failed", elapsed)
            logger.error(f"  ✗ {name} FAILED after {elapsed:.1f}s: {e}\n{traceback.format_exc()}")
            return False

    def run(self):
        """Run all phases respecting dependency order. Parallel where possible."""
        completed = set()
        failed = set()

        # Group phases into waves: all phases whose deps are satisfied
        remaining = list(self._order)
        with ThreadPoolExecutor(max_workers=8) as pool:
            while remaining:
                ready = [
                    n for n in remaining
                    if all(d in completed for d in self.phases[n].deps)
                    and not any(d in failed for d in self.phases[n].deps)
                ]
                blocked = [
                    n for n in remaining
                    if any(d in failed for d in self.phases[n].deps)
                ]
                for n in blocked:
                    logger.warning(f"  ⏭  {n} — skipped (dependency failed)")
                    remaining.remove(n)

                if not ready:
                    break  # nothing left can run

                futures = {pool.submit(self._run_one, n): n for n in ready}
                for future in as_completed(futures):
                    n = futures[future]
                    remaining.remove(n)
                    if future.result():
                        completed.add(n)
                    else:
                        failed.add(n)

        logger.info(f"\nPipeline done. ✓ {len(completed)} completed, ✗ {len(failed)} failed")
        if failed:
            logger.warning(f"Failed phases: {sorted(failed)}")
```
